Python路径函数

返回目录或者文件名

path=r’D:\file.txt’
os.path.split(path)[1] 或者 os.path.basename(path) #file.txt
os.path.split(path)[0] 或者 os.path.dirname(path) #D:\

os.path.splitext(path) #(‘D:\file’,’.txt’)
os.path.splitext(path)[0] #D:\file
os.path.splitext(path)[1] #.txt

os.path.exists(path)
os.path.isfile(path)
os.path.isdir(path) #判断目录是否为目录

os.path.join(path1,path2) #拼接路径，以系统分隔符(os.sep)
os.path.normcase(path) #规范化路径，将反斜杆转为斜杆，字母转为小写

os.path.walk(path,visit,arg) #遍历目录及子目录
path:待遍历根目录
visit(arg,dirname,names)遍历目录的函数dirname目录名，names目录下文件名（也包含目录）字列表

#####
os.getcwd() #得到当前工作目录，即目前Python脚本工作目录路径
os.listdir() #返回指定目录下所有文件和目录名称
os.remove() #删除文件
os.removedirs() #删除多个目录
os.path.isabs() #判断是否是绝对路径
os.getenv()与os.putenv() #读取和设置环境变量
os.linesep #返回当前系统的换行符号 window ‘\r\n’ Linux ‘\n’ Mac使用’\r’
os.name #对window,它是nt,而对于Linux/Unix用户，它是posix
os.rename(old,new) #重命名
os.makedirs(r’D:\text\text.txt’) #创建多级目录
os.mkdir(‘text’) #创建单目录
os.stat(file) #获取文件属性
os.exit()
os.path.getsize(filename) #获取文件大小
